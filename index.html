// set start date as May 11, 2023

const startDate = new Date("May 11, 2023 00:00:00");


// set the amount to accrue per day in EUR

const amountPerDay = 50000;


function updateCounter() {

  // calculate the number of seconds elapsed since the start date

  const elapsedSeconds = Math.floor((new Date() - startDate) / 1000);


  // calculate the total amount accrued

  const totalAmount = elapsedSeconds * amountPerDay / (60 * 60 * 24);


  // update the counter value

  const countElement = document.getElementById("count");

  countElement.textContent = totalAmount.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2});

}


// update the counter every second

setInterval(updateCounter, 1000);
